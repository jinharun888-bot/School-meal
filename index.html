    <script>

        const schoolInput = document.getElementById('schoolInput');

        const dateInput = document.getElementById('dateInput');

        const searchBtn = document.getElementById('searchBtn');

        const errorMessage = document.getElementById('errorMessage');

        const noMealSection = document.getElementById('noMeal');

        const loading = document.getElementById('loading');

        const calDisplay = document.getElementById('calDisplay');

        const menuList = document.getElementById('menuList');

        const starRatingContainer = document.getElementById('starRating');

        const toggleListBtn = document.getElementById('toggleListBtn');

        const titleText = document.getElementById('titleText');

        const settingsBtn = document.getElementById('settingsBtn');

        const themeModal = document.getElementById('themeModal');

        const closeModal = document.getElementById('closeModal');

        const modalOverlay = document.getElementById('modalOverlay');

        

        let currentRating = 0;

        let currentMenuData = [];

        let currentIngredientData = [];

        let isShowingIngredients = false;


        // 테마 설정

        function setTheme(theme) {

            const body = document.body;

            body.classList.remove('theme-default', 'theme-blue', 'theme-green', 'theme-pink', 'theme-amber', 'theme-purple', 'theme-dark');

            body.classList.add(`theme-${theme}`);

            localStorage.setItem('user-theme', theme);

            themeModal.classList.add('hidden');

        }


        window.onload = () => {

            const today = new Date().toISOString().split('T')[0];

            dateInput.value = today;

            updateDateDisplay(new Date());

            const savedTheme = localStorage.getItem('user-theme') || 'default';

            setTheme(savedTheme);

            initStars();

        };


        function updateDateDisplay(dateObj) {

            document.getElementById('currentDateDisplay').textContent = dateObj.toLocaleDateString('ko-KR', { weekday: 'long', year: 'numeric', month: 'long', day: 'numeric' });

        }


        // 날짜 변경 함수 (왼쪽: 어제 -1, 오른쪽: 내일 +1)

        function changeDate(days) {

            const currentDate = new Date(dateInput.value);

            currentDate.setDate(currentDate.getDate() + days);

            const newDateStr = currentDate.toISOString().split('T')[0];

            dateInput.value = newDateStr;

            updateDateDisplay(currentDate);

            getMeal();

        }


        settingsBtn.onclick = () => themeModal.classList.remove('hidden');

        closeModal.onclick = () => themeModal.classList.add('hidden');

        modalOverlay.onclick = () => themeModal.classList.add('hidden');


        async function getMeal() {

            const schoolName = schoolInput.value.trim();

            const date = dateInput.value.replace(/-/g, '');


            if (!schoolName) {

                errorMessage.classList.remove('hidden');

                schoolInput.classList.add('border-red-500', 'error-shake');

                setTimeout(() => schoolInput.classList.remove('error-shake'), 500);

                return;

            }


            errorMessage.classList.add('hidden');

            schoolInput.classList.remove('border-red-500');

            loading.classList.remove('hidden');

            resultSection.classList.add('hidden');

            noMealSection.classList.add('hidden');


            try {

                // 1. 학교 정보 검색 (학교 코드 획득)

                const schoolRes = await fetch(`https://open.neis.go.kr/hub/schoolInfo?Type=json&SCHUL_NM=${encodeURIComponent(schoolName)}`);

                const schoolData = await schoolRes.json();


                if (!schoolData.schoolInfo) {

                    loading.classList.add('hidden');

                    errorMessage.textContent = "학교를 찾을 수 없습니다.";

                    errorMessage.classList.remove('hidden');

                    return;

                }


                const schoolCode = schoolData.schoolInfo[1].row[0].SD_SCHUL_CODE;

                const atptCode = schoolData.schoolInfo[1].row[0].ATPT_OFCDC_SC_CODE;


                // 2. 급식 정보 검색

                const mealRes = await fetch(`https://open.neis.go.kr/hub/mealServiceDietInfo?Type=json&ATPT_OFCDC_SC_CODE=${atptCode}&SD_SCHUL_CODE=${schoolCode}&MLSV_YMD=${date}`);

                const mealData = await mealRes.json();


                if (!mealData.mealServiceDietInfo) {

                    loading.classList.add('hidden');

                    noMealSection.classList.remove('hidden');

                    return;

                }

                

                const meal = mealData.mealServiceDietInfo[1].row[0];

                

                // 데이터 파싱 (알레르기 숫자 제거 등)

                currentMenuData = meal.DDISH_NM.split('<br/>').map(item => item.replace(/\([0-9.]+\)/g, '').trim());

                currentIngredientData = meal.ORGR_INFO ? meal.ORGR_INFO.split('<br/>').map(item => item.trim()) : ["식재료 정보가 없습니다."];

                

                isShowingIngredients = false;

                renderContent(); // 급식 내용 렌더링

                

                calDisplay.textContent = meal.CAL_INFO;


                resultSection.classList.remove('hidden');

                resetRating(); // 날짜 바뀔 때 별점 초기화


            } catch (err) {

                console.error(err);

                loading.classList.add('hidden');

                errorMessage.textContent = "정보를 가져오는 중 오류가 발생했습니다.";

                errorMessage.classList.remove('hidden');

            } finally {

                loading.classList.add('hidden');

            }

        }


        function renderContent() {

            const data = isShowingIngredients ? currentIngredientData : currentMenuData;

            titleText.textContent = isShowingIngredients ? "식재료 상세 정보" : "식단 구성 메뉴";

            toggleListBtn.textContent = isShowingIngredients ? "메뉴 보기" : "식재료 보기";


            menuList.innerHTML = data.map(item => `

                <div class="flex items-center space-x-3 py-1.5 animate-in fade-in slide-in-from-left-2 duration-300">

                    <span class="w-1.5 h-1.5 rounded-full bg-current opacity-40"></span>

                    <span class="text-base font-medium">${item}</span>

                </div>

            `).join('');

        }


        toggleListBtn.onclick = () => {

            isShowingIngredients = !isShowingIngredients;

            renderContent();

        };


        searchBtn.onclick = getMeal;


        function initStars() {

            starRatingContainer.innerHTML = '';

            for (let i = 1; i <= 5; i++) {

                const star = document.createElement('button');

                star.innerHTML = `<svg class="w-6 h-6" fill="currentColor" viewBox="0 0 20 20"><path d="M9.049 2.927c.3-.921 1.603-.921 1.902 0l1.07 3.292a1 1 0 00.95.69h3.462c.969 0 1.371 1.24.588 1.81l-2.8 2.034a1 1 0 00-.364 1.118l1.07 3.292c.3.921-.755 1.688-1.54 1.118l-2.8-2.034a1 1 0 00-1.175 0l-2.8 2.034c-.784.57-1.838-.197-1.539-1.118l1.07-3.292a1 1 0 00-.364-1.118L2.98 8.72c-.783-.57-.38-1.81.588-1.81h3.461a1 1 0 00.951-.69l1.07-3.292z"></path></svg>`;

                star.className = "text-gray-300 transition-all duration-200 hover:scale-110";

                star.onclick = () => setRating(i);

                starRatingContainer.appendChild(star);

            }

        }


        function setRating(rating) {

            currentRating = rating;

            const stars = starRatingContainer.children;

            for (let i = 0; i < 5; i++) {

                if (i < rating) {

                    stars[i].classList.add('star-active');

                    stars[i].classList.remove('text-gray-300');

                } else {

                    stars[i].classList.remove('star-active');

                    stars[i].classList.add('text-gray-300');

                }

            }
